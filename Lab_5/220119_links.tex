\section*{Architecture Summary}

\begin{itemize}[leftmargin=*]
    \item \textbf{Shallow NN:} Input $\rightarrow$ 1 hidden layer (ReLU/Sigmoid/Tanh) $\rightarrow$ Output
    \item \textbf{Deep NN:} Input $\rightarrow$ 3 hidden layers (ReLU + BatchNorm + Dropout) $\rightarrow$ Output
    \item \textbf{Dataset:} Teens Mental Health Dataset (1\,200 rows, 12 features)
    \item \textbf{Imbalance handling:} Weighted Binary Cross-Entropy loss
\end{itemize}
