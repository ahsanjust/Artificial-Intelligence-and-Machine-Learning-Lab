\section*{Architecture Summary}

\begin{itemize}[leftmargin=*]
    \item \textbf{Shallow NN:} Input $\rightarrow$ 1 hidden layer (64 neurons, ReLU) $\rightarrow$ 3-class Output
    \item \textbf{Deep NN:} Input $\rightarrow$ 3 hidden layers (128/64/32, ReLU + BatchNorm + Dropout) $\rightarrow$ 3-class Output
    \item \textbf{Dataset:} AI Student Impact Dataset (50\,000 rows, 15 features)
    \item \textbf{Target:} Burnout\_Risk\_Level (Low / Medium / High) -- multi-class
    \item \textbf{Loss:} Weighted Cross-Entropy (class-weighted)
    \item \textbf{Metrics:} Weighted Precision, Recall, F1, AUC-ROC (One-vs-Rest)
\end{itemize}
